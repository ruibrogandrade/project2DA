45// Make a value go "up the tree" until it reaches its position
46template <class K, class V>
47void MaxHeap<K,V>::upHeap(int i) {
48 while (i>1 && a[i].value < a[PARENT(i)].value) { // while pos smaller than parent, keep swapping to upper position
49 swap(i, PARENT(i));
50 i = PARENT(i);
51 }
52}
53
54// Make a value go "down the tree" until it reaches its position
55template <class K, class V>
56void MaxHeap<K,V>::downHeap(int i) {
57 while (LEFT(i) <= size) { // while within heap limits
58 int j = LEFT(i);
59 if (RIGHT(i)<=size && a[RIGHT(i)].value < a[j].value) j = RIGHT(i); // choose smaller child
60 if (a[i].value < a[j].value) break; // node already smaller than children, stop
61 swap(i, j); // otherwise, swap with smaller child
62 i = j;
63 }
64}

66// Swap two positions of the heap (update their positions)
67template <class K, class V>
68void MaxHeap<K,V>::swap(int i1, int i2) {
69 Node tmp = a[i1]; a[i1] = a[i2]; a[i2] = tmp;
70 pos[a[i1].key] = i1;
71 pos[a[i2].key] = i2;
72}

93// Insert (key, value) on the heap
94template <class K, class V>
95void MaxHeap<K,V>::insert(const K& key, const V& value) {
96 if (size == maxSize) return; // heap is full, do nothing
97 if (hasKey(key)) return; // key already exists, do nothing
98 a[++size] = {key, value};
99 pos[key] = size;
100 upHeap(size);
101}

103// Increase value of key to the indicated value
104template <class K, class V>
105void MaxHeap<K,V>::increaseKey(const K &key, const V &value) {
106 if (!hasKey(key)) return; // key does not exist, do nothing
107 int i = pos[key];
108 if (value > a[i].value) return; // value would decrease, do nothing
109 a[i].value = value;
110 upHeap(i);
111}

113// remove and return key with smaller value
114template <class K, class V>
115K MaxHeap<K,V>::removeMax() {
116 if (size == 0) return KEY_NOT_FOUND;
117 K min = a[1].key;
118 pos.erase(min);
119 a[1] = a[size--];
120 downHeap(1);
121 return min;
122}
